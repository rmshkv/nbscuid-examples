ocean_stats = []
for c, l, p in zip(casename,label, ocn_path):
  ds = xr.open_dataset(p+'{}_ocean.stats.nc'.format(c))
  stats_monthly = ds.resample(time="1M", 
                               closed='left',
                               keep_attrs='True').mean('time', keep_attrs=True)
  ocean_stats.append(stats_monthly)
